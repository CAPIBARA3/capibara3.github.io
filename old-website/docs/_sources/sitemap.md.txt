## Documentation Network (Public)

Explore the public documentation as an interactive network. Click any node to open that page.

```{raw} html
<style>
	#docs-graph {
		height: 520px;
		border: 1px solid #d6d6d6;
		border-radius: 10px;
		background: #fafafa;
	}
	#docs-graph .graph-loading {
		padding: 16px;
		color: #555;
	}
</style>

<div id="docs-graph">
	<div class="graph-loading">Loading documentation graph...</div>
</div>

<script src="https://unpkg.com/vis-network/standalone/umd/vis-network.min.js"></script>
<script src="_static/docs-graph.js"></script>
<script>
(() => {
	const container = document.getElementById("docs-graph");
	if (!container || typeof vis === "undefined" || !window.DOCS_GRAPH) {
		return;
	}

	const data = window.DOCS_GRAPH;
	const nodes = new vis.DataSet(data.nodes || []);
	const edges = new vis.DataSet(data.edges || []);
	const options = {
		layout: {
			improvedLayout: true,
		},
		physics: {
			stabilization: true,
			barnesHut: {
				gravitationalConstant: -28000,
				springLength: 120,
				springConstant: 0.04,
			},
		},
		nodes: {
			shape: "dot",
			size: 10,
			font: {
				size: 13,
			},
		},
		edges: {
			color: {
				color: "#9aa4ad",
				highlight: "#3d6fb6",
			},
			smooth: {
				type: "dynamic",
			},
		},
		groups: {
			directory: {
				shape: "box",
				color: {
					background: "#f0f4ff",
					border: "#5171a6",
				},
				font: {
					color: "#29406b",
				},
			},
			page: {
				shape: "dot",
				color: {
					background: "#ffe3c6",
					border: "#d97832",
				},
				font: {
					color: "#6a3d14",
				},
			},
		},
		interaction: {
			hover: true,
			tooltipDelay: 120,
		},
	};

	const network = new vis.Network(container, { nodes, edges }, options);

	network.on("click", (params) => {
		if (!params.nodes || params.nodes.length === 0) {
			return;
		}
		const node = nodes.get(params.nodes[0]);
		if (node && node.url) {
			window.location.href = node.url;
		}
	});
})();
</script>
```
